const diamond = (n) => {
  if (n % 2 === 0 || n < 0) return null  
  
  const middle = Math.round(n / 2)
  let w; 
  let diam = '';
  
  for (let i = 1; i <= n; i++) {
    w = i <= middle ? i : n - i + 1;
    diam += ' '.repeat(middle - w) + '*'.repeat(2 * w - 1) + '\n';
  }
  return diam
}
